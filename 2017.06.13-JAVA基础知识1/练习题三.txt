1:基本小题目
	int a = 10;
	int b = 10;
	int c = 10;

	a = b++;   a = 10;b = 11
	c = --a;   a = 9 ;c = 9
	b = ++a;   a = 10;b = 10
	a = c--;   a = 9 ;c = 8
	请分别计算出a,b,c的值

a = 9        b = 10       c = 8
